# 5/22/24: Fetch clan stats from Clash Royale API and write as Excel to Discord channel.
The Debian VM guest has a cron job that runs a python script in a container.

ssh <debian-vm-host>
sudo -i
cd /mnt/get_clan_stats
root@debian-vm:/mnt/get_clan_stats# 

docker image ls && docker container ls
docker system prune --force

# Build the image:
docker-compose build

# Clean up, but don't delete danglers:
docker system prune --force

# Run one time now, deleting container upon exit:
docker-compose run --rm get-clan-stats

# Put its invocation into the crontab file for root:
crontab -e
# 5/22/24: Write clan stats to Discord:
23 6,18 * * * cd /mnt/get_clan_stats && /usr/bin/docker-compose run --rm get-clan-stats >/dev/null 2>&1

less /var/log/syslog

Run interactively while debugging:
root@debian-vm:/mnt/get_clan_stats# 
docker system prune --force && docker-compose build && docker-compose run --rm get-clan-stats

# From MacBook, back up the files from Debian VM guest to OneDrive, preserving timestamps:
$ scp -p walt@<debian-vm-host>:"/mnt/get_clan_stats/*" ~/OneDrive/Documents/AppDev/Supercell/Clash\ Royale/working
